cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(light_point glfw.icns light.cpp)

target_link_libraries(light_point PRIVATE stbimage)

target_include_directories(light_point PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET light_point POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/light_point.app/Contents/Resources
)
