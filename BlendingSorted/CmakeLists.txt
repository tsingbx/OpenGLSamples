cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(blending_sorted glfw.icns blending_sorted.cpp)

target_link_libraries(blending_sorted PRIVATE stbimage)

target_include_directories(blending_sorted PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET blending_sorted POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/blending_sorted.app/Contents/Resources
)
