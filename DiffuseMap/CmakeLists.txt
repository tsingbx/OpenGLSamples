cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(diffuse_map glfw.icns diffuse_map.cpp)

target_link_libraries(diffuse_map PRIVATE stbimage)

target_include_directories(diffuse_map PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET diffuse_map POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/diffuse_map.app/Contents/Resources
)
