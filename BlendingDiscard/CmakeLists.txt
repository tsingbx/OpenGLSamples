cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(blending_discard glfw.icns blending_discard.cpp)

target_link_libraries(blending_discard PRIVATE stbimage)

target_include_directories(blending_discard PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET blending_discard POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/blending_discard.app/Contents/Resources
)
