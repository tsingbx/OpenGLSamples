cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(camera glfw.icns camera.cpp)

target_link_libraries(camera PRIVATE stbimage)

target_include_directories(camera PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET camera POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/camera.app/Contents/Resources
)
