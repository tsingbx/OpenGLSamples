cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(cubemaps_mapping glfw.icns skybox.cpp)

target_link_libraries(cubemaps_mapping PRIVATE stbimage)

target_include_directories(cubemaps_mapping PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET cubemaps_mapping POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/cubemaps_mapping.app/Contents/Resources
)
