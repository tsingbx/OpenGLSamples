cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(coordnate2 glfw.icns coordnate.cpp)

target_link_libraries(coordnate2 PRIVATE stbimage)

target_include_directories(coordnate2 PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET coordnate2 POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/coordnate2.app/Contents/Resources
)
