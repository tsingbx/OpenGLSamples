cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(cubemaps_skybox glfw.icns skybox.cpp)

target_link_libraries(cubemaps_skybox PRIVATE stbimage)

target_include_directories(cubemaps_skybox PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET cubemaps_skybox POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/cubemaps_skybox.app/Contents/Resources
)
