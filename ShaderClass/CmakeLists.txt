cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(shader_class glfw.icns shader.cpp)

target_include_directories(shader_class BEFORE PRIVATE "${CMAKE_SOURCE_DIR}/deps")

add_custom_command(
    TARGET shader_class POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/shader_class.app/Contents/Resources
)
