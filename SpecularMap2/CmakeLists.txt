cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(specular_map2 glfw.icns specular_map.cpp)

target_link_libraries(specular_map2 PRIVATE stbimage)

target_include_directories(specular_map2 PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET specular_map2 POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/specular_map2.app/Contents/Resources
)
