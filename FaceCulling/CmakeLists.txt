cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(face_culling glfw.icns face_culling.cpp)

target_link_libraries(face_culling PRIVATE stbimage)

target_include_directories(face_culling PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET face_culling POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/face_culling.app/Contents/Resources
)
