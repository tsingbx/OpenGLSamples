cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(geometry_shader_exploding glfw.icns geometry_shader_exploding.cpp)

target_link_libraries(geometry_shader_exploding PRIVATE stbimage assimp)

target_link_directories(geometry_shader_exploding PRIVATE /opt/homebrew/lib/)

target_include_directories(geometry_shader_exploding PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage /opt/homebrew/include/)

add_custom_command(
    TARGET geometry_shader_exploding POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/geometry_shader_exploding.app/Contents/Resources
)
