cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(geometry_shader_houses glfw.icns geometry_shader_houses.cpp)

target_link_libraries(geometry_shader_houses PRIVATE stbimage)

target_include_directories(geometry_shader_houses PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET geometry_shader_houses POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/geometry_shader_houses.app/Contents/Resources
)
