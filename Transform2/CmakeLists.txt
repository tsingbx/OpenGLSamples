cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(transform2 glfw.icns transform.cpp)

target_link_libraries(transform2 PUBLIC stbimage)

target_include_directories(transform2 PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET transform2 POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/transform2.app/Contents/Resources
)
