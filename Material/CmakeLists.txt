cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(material glfw.icns material.cpp)

target_link_libraries(material PRIVATE stbimage)

target_include_directories(material PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET material POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/material.app/Contents/Resources
)
