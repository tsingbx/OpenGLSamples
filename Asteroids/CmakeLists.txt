cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(asteroids glfw.icns asteroids.cpp)

target_link_libraries(asteroids PRIVATE stbimage assimp)

target_link_directories(asteroids PRIVATE /opt/homebrew/lib/)

target_include_directories(asteroids PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage /opt/homebrew/include/)

add_custom_command(
    TARGET asteroids POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/asteroids.app/Contents/Resources
)
