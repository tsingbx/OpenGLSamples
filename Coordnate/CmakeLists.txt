cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(coordnate glfw.icns coordnate.cpp)

target_link_libraries(coordnate PUBLIC stbimage)

target_include_directories(coordnate PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET coordnate POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/coordnate.app/Contents/Resources
)
