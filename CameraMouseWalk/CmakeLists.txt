cmake_minimum_required(VERSION 3.16...3.28 FATAL_ERROR)

AddExe(camera_mouse_walk glfw.icns camera.cpp)

target_link_libraries(camera_mouse_walk PRIVATE stbimage)

target_include_directories(camera_mouse_walk PRIVATE ${CMAKE_SOURCE_DIR}/deps ${CMAKE_SOURCE_DIR}/StbImage)

add_custom_command(
    TARGET camera_mouse_walk POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
    ${CMAKE_CURRENT_SOURCE_DIR}/resources/
    ${CMAKE_CURRENT_BINARY_DIR}/camera_mouse_walk.app/Contents/Resources
)
